class Solution {
//941. 有效的山脉数组
    public boolean validMountainArray(int[] A) {

        int i = 0;
        while (i < (A.length -1) && A[i] < A[i+1]) {
            i++;
        }
        if (i == 0 || i== (A.length-1)) return false;

        while (i< (A.length -1) && A[i] > A[i+1]) {
            i++;
        }
        return i== A.length-1;
    }
}
